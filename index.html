    Dependency injection (DI)

    
// home.service.ts
@Injectable({
	providedIn: 'root'
})
export class HomeService {
	activeUser = 'Greger';
	constructor() { }
}
					
    
// home.component.ts
@Component({
	...
})
export class HomeComponent implements OnInit {
	constructor(private homeService: HomeService) { }
	ngOnInit() {
		console.log(this.homeService.activeUser);
	}
}
    Uses DI to decrease tight coopeling and make testing easer as DI service can be swapped for mock
