    Routing

    Eager loading vs Lazy loading

    
// Eager (default)
const routes: Routes = [{
	path: 'home', // localhost:4200/home
	component: HomeComponent,
}]; 
export const HomeRoutes = RouterModule.forChild(routes);
					
    


					
    
// Lazy
const routes: Routes = [{
	path: 'home', // localhost:4200/home
	loadChildren: () => import('./home.module').then(m => m.HomeModule)
}]; 
export const HomeRoutes = RouterModule.forChild(routes);
